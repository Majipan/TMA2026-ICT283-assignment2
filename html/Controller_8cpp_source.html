361 // Remove duplicates from Vector from a sorted vector
362 void Controller::RemoveDuplicateRecords(WeatherData& records) const
363 {
364  // Safety check for when vector has only 1 or less record
365  if(records.Size() <= 1)
366  {
367  return;
368  }
369 
370  // tracks the last unique record
371  int writeIndex = 1;
372 
373  // iterate through the vector to look for duplicates
374  for(int readIndex = 1; readIndex < records.Size(); ++readIndex)
375  {
376  // compares current record with last unique record and skip if duplicate
377  if(!(records[readIndex] == records[writeIndex - 1]))
378  {
379  // if there are already records skipped, write new unique record into last available slot
380  if(writeIndex != readIndex)
381  {
382  records[writeIndex] = records[readIndex];
383  }
384 
385  // move unique boundary forward
386  ++writeIndex;
387  }
388  }
389 
390  // purge excess records
391  while(records.Size() > writeIndex)
392  {
393  records.Delete(records.Size() - 1);
394  }
395 }

399 void Controller::BalancedInsert(const WeatherData& records,
400  int left,
401  int right,
402  WeatherTree& tree) const
403 {
404  if(left > right)

952 // Helper method to calculate SPCC
953 bool Controller::CalculateSPCC(const WeatherData& records,
954  WeatherGetter xGetter,
955  WeatherValidGetter xValid,
956  bool xIsSolar,
957  WeatherGetter yGetter,
958  WeatherValidGetter yValid,
959  bool yIsSolar,
960  double& result) const
961 {
962  int count = 0;
963 
964  double sumX = 0.0;
965  double sumY = 0.0;
966  double sumXY = 0.0;
967  double sumX2 = 0.0;
968  double sumY2 = 0.0;
969 
970  for(int i = 0; i < records.Size(); ++i)
971  {
972  if(IsValidSPCCValue(records[i], xGetter, xValid, xIsSolar) &&
973  IsValidSPCCValue(records[i], yGetter, yValid, yIsSolar))
974  {
975  double x = (records[i].*xGetter)();
976  double y = (records[i].*yGetter)();
977 
978  sumX += x;
979  sumY += y;
980  sumXY += x * y;
981  sumX2 += x * x;
982  sumY2 += y * y;
983 
984  ++count;
985  }
986  }
987 
988  if(count < 2)
989  {
990  return false;
991  }
992 
993  double numerator = count * sumXY - sumX * sumY;
994 
995  double denominatorLeft = count * sumX2 - sumX * sumX;
996  double denominatorRight = count * sumY2 - sumY * sumY;
997 
998  if(denominatorLeft <= 0.0 || denominatorRight <= 0.0)
999  {
1000  return false;
1001  }
1002 
1003  result = numerator / sqrt(denominatorLeft * denominatorRight);
1004 
1005  return true;
1006 }
